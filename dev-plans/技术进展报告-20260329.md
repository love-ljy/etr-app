## ✅ 核心功能验证

### 1. USDT 入金流程

**PRD 要求**:
```
用户 100 USDT → 合约从 LP 池买 ETR → 扣 3% 滑点 → 97 USDT 等值 ETR → 自动质押
```

**当前实现**: `StakingPoolV2.stake()`

```solidity
// ✅ 已实现
function stake(uint256 usdtAmount, address referrer) external returns (uint256) {
    // 1. 计算应得 ETR 份额
    uint256 etrAmount = usdtAmount * 1e18 / etrPriceUSD;
    
    // 2. 转移 USDT 到合约
    usdtToken.safeTransferFrom(msg.sender, address(this), usdtAmount);
    
    // 3. 创建质押记录（自动锁仓 50 天）
    stakes[stakeId] = StakeRecord({
        principal: etrAmount,
        startTime: block.timestamp,
        unlockTime: block.timestamp + 50 days,
        active: true
    });
}
```

**滑点扣除**: `ETRToken._transfer()` 中实现，买入固定 3%

**结论**: ✅ **符合 PRD 要求**

### 3. 动态滑点保护

**PRD 要求**:
- 买入固定 3%
- 卖出基础 3%
- 价格跌幅≥10%/20%/30% 滑点自动提升
- 每日自动恢复 5%

**当前实现**: `SlippageController.sol`

```solidity
// ✅ 价格跌幅检测
function _calculateAndAdjustSlippage() internal {
    uint256 drop = ((openPrice - currentPrice) * BPS_DENOMINATOR) / openPrice;
    
    if (drop >= DROP_40_PERCENT) newSellFee = SLIPPAGE_40_PERCENT;  // 40%
    else if (drop >= DROP_30_PERCENT) newSellFee = SLIPPAGE_30_PERCENT;  // 30%
    else if (drop >= DROP_20_PERCENT) newSellFee = SLIPPAGE_20_PERCENT;  // 20%
    else if (drop >= DROP_10_PERCENT) newSellFee = SLIPPAGE_10_PERCENT;  // 10%
}

// ✅ 每日恢复 5%
function _tryRecoverSlippage() internal {
    if (block.timestamp < slippageState.lastRecoveryTime + RECOVERY_INTERVAL) {
        return;
    }
    
    // 每日恢复 5% (500 基点)
    if (oldFee > BASE_SELL_FEE + DAILY_RECOVERY_RATE) {
        newFee = oldFee - DAILY_RECOVERY_RATE;
    } else {
        newFee = BASE_SELL_FEE;
    }
}
```

**结论**: ✅ **完全符合 PRD 要求**

function dropLPRole(address lpTokenAddress) external onlyOwner {

    uint256 balance = lpToken.balanceOf(address(this));
    if (balance > 0) {

**问题**: 当前 ETR 直接存入进入复利池，应该进入质押池

**解决方案**:

```solidity
// 在 StakingPoolV2.sol 中新增函数
function stakeETR(uint256 etrAmount, address referrer) external returns (uint256) {
    require(etrAmount > 0, "StakingPoolV2: amount must be greater than 0");
    
    // 1. 转移 ETR 到合约
    etrToken.safeTransferFrom(msg.sender, address(this), etrAmount);
    
    // 2. 创建质押记录
    uint256 stakeId = nextStakeId++;
    
    stakes[stakeId] = StakeRecord({
        stakeId: stakeId,
        owner: msg.sender,
        stakedUSDT: etrAmount * etrPriceUSD / 1e18,  // 转换为 USDT 价值
        principal: etrAmount,
        originalPrincipal: etrAmount,
        startTime: block.timestamp,
        unlockTime: block.timestamp + config.lockPeriod,
        lastClaimTime: block.timestamp,
        totalClaimed: 0,
        dailyYieldRate: currentYieldRate,
        active: true
    });
    
    // 3. 更新用户账户和推荐关系
    // ... (与 stake() 相同逻辑)
    
    emit Staked(msg.sender, stakeId, 0, etrAmount, unlockTime, block.timestamp);
    
    return stakeId;
}
```

```typescript
export function handleStaked(event: Staked): void {
  let user = User.load(event.params.user.toHexString());
  if (!user) {
    user = new User(event.params.user.toHexString());
    user.address = event.params.user;
    user.totalStaked = BigInt.zero();
    user.totalClaimed = BigInt.zero();
    user.stakeCount = 0;
    user.createdAt = event.block.timestamp;
  }
  
  let stake = new StakeRecord(event.params.stakeId.toString());
  stake.stakeId = event.params.stakeId;
  stake.user = user.id;
  stake.amount = event.params.etrAmount;
  stake.startTime = event.block.timestamp;
  stake.unlockTime = event.params.unlockTime;
  stake.status = StakeStatus.ACTIVE;
  
  stake.save();
  user.save();
}
```
